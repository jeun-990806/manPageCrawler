from selenium import webdriver
import requests
from bs4 import BeautifulSoup

driverPath = '/home/jeun/PycharmProjects/summerProj/geckodriver'
browser = webdriver.Firefox(executable_path=driverPath)
browser.get('https://man7.org/linux/man-pages/dir_section_3.html')

menus = browser.find_elements_by_css_selector('a')

target = None
for m in menus:
    if m.text == 'printf(3)':
        target = m

target.click()

texts = browser.find_elements_by_css_selector('pre')
for text in texts:
    print(text.text)

url = 'https://man7.org/linux/man-pages/man3/printf.3.html'
response = requests.get(url)

if response.status_code == 200:
    html = response.text
    soup = BeautifulSoup(html, 'html.parser')
    texts = soup.select('pre')
    for text in texts:
        print(text.get_text())
